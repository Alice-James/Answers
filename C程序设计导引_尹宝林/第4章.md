{

	i = 0;

		sign = (i % 2 == 0)? 1 : -1;

		for(j = 0, n = 1.0; j < k; j++)

		d = 1.0 / (n * k);
		sa += d * sign;
		i++;

	i = 0;

		sign = (i % 2 == 0)? 1 : -1;
		k = 2 * i + 1;

			n *= 239;
		d = 1.0 / (n * k);
		sb += d * sign;
		i++;
	}while(d > eps);
	
	printf("%8f\n", 16 * sa - 4 * sb);

}

int main()
{
	int i, j, sign, sum;
	
	for(i = 2, sum = 0; i <= MAX; i++){
		for(j = 2, sign = 1; j <= i / 2; j++){
			if(i % j == 0 && i != j){
				sign = 0;
				break;
			}
		}
		if(sign){
			sum++;
			printf("%d", i);
			if(sum % 10 != 0)

				putchar('\n');
		}

#include <stdio.h>

bool runnian(int y);
int xingqi(int y, int m, int d);

bool runnian(int y)
{
	if((y % 4 == 0 && y % 100 != 0) || y % 400 == 0)
		return 1;
	else
		return 0;
}

int xingqi(int Y, int M, int D)
{
	int y, m, total;
	if(Y >= 2000){
		total = 0;
		for(y = 2000; y < Y; y++){
			if(runnian(y))
				total += 366;
			else
				total += 365;
		}
		for(m = 1; m < M; m++){
			switch(m){
				case 1: case 3: case 5: case 7: case 8: case 10: case 12:
					total += 31;
					break;
				case 4: case 6: case 9: case 11:
					total += 30;
					break;
				case 2:
					if((Y % 4 == 0 && Y % 100 != 0) || Y % 400 == 0)
						total += 29;
					else
						total += 28;
			}
		}
		total += D - 1;
		return (total % 7 <= 1)? total % 7 + 6 : total % 7 - 1;
	}
	else{
		total = 1;
		for(y = 1999; y > Y; y--){
			if(runnian(y))
				total += 366;
			else
				total += 365;
		}
		for(m = 12; m > M; m--){
			switch(m){
				case 1: case 3: case 5: case 7: case 8: case 10: case 12:
					total += 31;
					break;
				case 4: case 6: case 9: case 11:
					total += 30;
					break;
				case 2:
					if(runnian(Y))
						total += 29;
					else
						total += 28;
			}
		}
		switch(M){
			case 1: case 3: case 5: case 7: case 8: case 10: case 12:
				total += 31 - D;
				break;
			case 4: case 6: case 9: case 11:
				total += 30 - D;
				break;
			case 2:
				if(runnian(Y))
					total += 29 - D;
				else
					total += 28 - D;
		}
		return (total % 7 == 6)? 7 : 6 - total % 7;
	}
}	

int main()
{
	int y1, m1, d1, n, y2, m2, d2, x, y_days, m_days;
	
	scanf("%d%d%d%d", &y1, &m1, &d1, &n);
	y2 = y1;
	m2 = m1;
	d2 = d1;
	while(n > 0){
		//确定年份为y2或（y2 - 1）
		if(runnian(y2 - 1))
			y_days = 366;
		else
			y_days = 365;
		if(n >= y_days){
			n -= y_days;
			y2--;
		}
		else
			break;
	}
	while(n > 0){
		//确定月份为m2或m2前一个月
		switch(m2 - 1){
			case 1: case 3: case 5: case 7: case 8: case 10: case 0:
				m_days = 31;

				break;

				m2--;
